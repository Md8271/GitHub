# git useful commands(cli) :-

1. git init
2. git add f1.md
3. git rm --cached f1.md
4. git restore --staged filename.extension

# commit
1. git commit -m "commit message"
2. git commit --amend --no-edit
3. git commit --amend -m "new first commit"
4. git add . 

# rename a branch name
1. git branch -m branchName =respectfully
2. git branch -M branchName =forcefully

.gitignore file
